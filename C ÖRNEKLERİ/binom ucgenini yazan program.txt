 int fact( int n)/// recursive( kendini çagiran ) fonksiyon
 {
 if(n==0||n==1)
 return 1;
 else if(n>1)
 return n*fact(n-1);/// kendini çagirma
 else
 {
 printf("wrong factorial\n");
 return -1;
 }
 }
